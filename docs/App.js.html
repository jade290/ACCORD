    /** @module App */

import React, { Component } from 'react';
import { BrowserRouter as Router, Switch, Route } from 'react-router-dom';
import ls from 'local-storage';

import Navbar from './components/Navbar';
import Footer from './components/Footer';
import NotFound from './views/NotFound';
import Views from './views';

import CONFIG from './config/config.json';

import './style.less';

/**
* App component handles view routing and renders view components, the navbar and footer.
*/
class App extends Component {
    state = {
        user: {},
    };

    /** run the contained logic when the component first mounts */
    componentDidMount() {
        this.getUserInfo();
    }

    /**
    * retrieves geoaxis user information formo local storage.
    * this user information is stored by the script at './controllers/geoaxis.js'
    * when the user logs in
    */
    getUserInfo = () => {
        let user = ls.get('gxUser');
        if (!user) user = { given_name: '' };
        this.setState({ user });
    }

    /** returns a single react-router-dom route */
    renderRoute = route => (
        <Route
            key={route.name}
            path={route.path}
            exact={route.exact}
            render={routeProps => (
                <route.view {...routeProps} user={this.state.user} />
            )}
        />
    )

    /**
    * renders the app copmponent.  maps registered views to an array of react-router-dom
    * route components
    */
    render() {
        return (
            <Router>
                <div id="app">
                    <Navbar user={this.state.user} />
                    <div id="content">
                        <Switch>
                            {Views.map(r => this.renderRoute(r))}
                            <Route key="404" render={() => <NotFound />} />
                        </Switch>
                    </div>
                    {!CONFIG.useEcobar && <Footer />}
                </div>
            </Router>
        );
    }
}

/** export the component for use in other files */
export default App;
